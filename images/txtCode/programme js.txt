// Variables pour stocker les scores des joueurs
let player1RoundScore = 0;
let player1TotalScore = 0;
let player2RoundScore = 0;
let player2TotalScore = 0;

// Variable pour stocker le joueur actuel
let currentPlayer = 1;

const player1RoundScoreDisplay = document.getElementById("player1-round-score");
const player1TotalScoreDisplay = document.getElementById("player1-total-score");
const player2RoundScoreDisplay = document.getElementById("player2-round-score");
const player2TotalScoreDisplay = document.getElementById("player2-total-score");
const messageDisplay = document.getElementById("message");

// Références aux boutons
const newGameButton = document.getElementById("new-game-button");
const rollDiceButton = document.getElementById("roll-dice-button");
const holdButton = document.getElementById("hold-button");

// Écouter les événements de clic sur les boutons
newGameButton.addEventListener("click", newGame);
rollDiceButton.addEventListener("click", rollDice);
holdButton.addEventListener("click", hold());

// Fonction pour lancer un nouveau jeu
function newGame() {
  player1RoundScore = 0;
  player1TotalScore = 0;
  player2RoundScore = 0;
  player2TotalScore = 0;
  currentPlayer = 1;
  updateScores();
  messageDisplay.textContent = "Player 1's turn";
}

// Fonction pour lancer le dé
function rollDice() {
  let roll = Math.floor(Math.random() * 6) + 1;
  if (roll === 1) {
    // Passage au joueur suivant si 1 est obtenu
    currentPlayer === 1 ? currentPlayer = 2 : currentPlayer = 1;
    // Réinitialisation du score de round
    currentPlayer === 1 ? player1RoundScore = 0 : player2RoundScore = 0;
    updateScores();
    messageDisplay.textContent = "Player " + currentPlayer + " rolled a 1. Next player's turn.";
  } else {
    // Ajout du score de roll au score de round
    currentPlayer === 1 ? player1RoundScore += roll : player2RoundScore += roll;
    updateScores();
    messageDisplay.textContent = "Player " + currentPlayer + "'s turn";
  }
}
